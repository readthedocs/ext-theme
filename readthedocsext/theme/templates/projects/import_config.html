# See https://docs.readthedocs.io/en/stable/config-file/v2.html for details

  os: ubuntu-22.04
  tools:
    python: "3.11"
    # You can also specify other tool versions:
    # nodejs: "19"
    # rust: "1.64"
    # golang: "1.19"

# Build documentation in the "docs/" directory with Sphinx

   configuration: docs/conf.py
